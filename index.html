ตั้งค่าการเชื่อมต่อ Supabase

เชื่อมโยงระบบเข้ากับฐานข้อมูลคลาวด์ Supabase ของคุณเพื่อเก็บข้อมูลจริงและเข้าสู่ระบบด้วยผู้ใช้อื่นๆ

ไม่ได้เชื่อมต่อฐานข้อมูล Supabase

คำแนะนำในการเริ่มต้นตั้งค่า Table ใน Supabase

เพื่อให้แอปสามารถทำงานได้สมบูรณ์ กรุณาสร้าง Table ต่อไปนี้ใน Supabase SQL Editor ของคุณ:

-- 1. สร้างตารางลูกค้าสำหรับเก็บแต้มสะสม
CREATE TABLE customers (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(255) NOT NULL,
    phone VARCHAR(20) UNIQUE NOT NULL,
    points INT DEFAULT 0,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- 2. สร้างตารางการสั่งซื้อ
CREATE TABLE orders (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    total_amount DECIMAL(10,2) NOT NULL,
    discount DECIMAL(10,2) DEFAULT 0.00,
    points_earned INT DEFAULT 0,
    points_redeemed INT DEFAULT 0,
    customer_id UUID REFERENCES customers(id),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- 3. สร้างตารางรายการสินค้าที่ซื้อในแต่ละบิล
CREATE TABLE order_items (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    order_id UUID REFERENCES orders(id) ON DELETE CASCADE,
    product_name VARCHAR(255) NOT NULL,
    quantity INT NOT NULL,
    price DECIMAL(10,2) NOT NULL,
    category VARCHAR(50) NOT NULL
);

-- 4. ตั้งค่า RLS (Row Level Security) หรือปิดเพื่อทดสอบพัฒนา
-- รันคำสั่งนี้เพื่อเปิดใช้งานแบบไม่จำกัดชั่วคราวในการพัฒนา
ALTER TABLE customers DISABLE ROW LEVEL SECURITY;
ALTER TABLE orders DISABLE ROW LEVEL SECURITY;
ALTER TABLE order_items DISABLE ROW LEVEL SECURITY;

หลังเชื่อมต่อแล้ว ให้รีเฟรชหน้าเว็บอีกครั้งเพื่อเชื่อมโยงข้อมูล!
